def _is_valid_png(b64_string: str) -> bool:
    """Return True if b64_string decodes to a valid PNG byte sequence."""
    try:
        raw = base64.b64decode(b64_string)
        return raw[:4] == b'\x89PNG'
    except Exception:
        return False
