    ```python
    """
    reporters/monthly.py — Monthly Close builder (~2000 mots, 7 sections, ≥ 2 tableaux).

    Sections (REPT-03) :
      1. Month in Review              — narration ~250 mots
      2. Macro Backdrop               — tableau FRED + narration
      3. ETF Monthly Performance      — tableau ETF + narration
      4. Crypto Monthly               — tableau coins + narration
      5. PEA Monthly                  — tableau PEA + alerte éligibilité
      6. News & Themes                — bullets + narration thèmes
      7. Forward Look                 — narration ~250 mots
    """
    from __future__ import annotations
    from config import Config
    from reporters.base import synthesize_section, build_section, format_pct, format_currency
    from logging_setup import get_logger

    logger = get_logger(__name__)

    MONTHLY_SYSTEM_PROMPT = (
        "Tu es un analyste financier français senior. Réponds en français, ton sobre et factuel. "
        "Sections de ~250 mots pour la narration centrale, ~150 mots pour les commentaires de tableau. "
        "Pas de conseil financier explicite."
    )

    def _table(headers, rows):
        if not rows:
            return ""
        head = "| " + " | ".join(headers) + " |"
        sep = "|" + "|".join(["---"] * len(headers)) + "|"
        body = "\n".join("| " + " | ".join(r) + " |" for r in rows)
        return f"{head}\n{sep}\n{body}"

    def _month_in_review(data, config):
        meta = data.get("_meta", {}) or {}
        prompt = (
            "Rédige 'Month in Review' (~250 mots) — synthèse du mois écoulé pour un investisseur "
            f"particulier français. Sources OK: {meta.get('sources_ok', [])}. "
            f"Failed: {meta.get('sources_failed', [])}."
        )
        body = synthesize_section(prompt, config=config, system=MONTHLY_SYSTEM_PROMPT, max_tokens=600)
        return build_section("Month in Review", body)

    def _macro_backdrop(data, config):
        macro = data.get("macro") or {}
        if macro.get("source_failed"):
            return build_section("Macro Backdrop", "Données macro indisponibles ce mois.")
        series = macro.get("series", {})
        rows = [[k, str(s.get("value", "n/a")), str(s.get("date", "n/a"))]
                for k, s in series.items() if isinstance(s, dict)]
        table = _table(["Série", "Valeur", "Date"], rows)
        prompt = f"Commentaire macro mensuel ~150 mots. Séries: {list(series.keys())}."
        body = synthesize_section(prompt, config=config, system=MONTHLY_SYSTEM_PROMPT)
        return build_section("Macro Backdrop", f"{table}\n\n{body}" if table else body)

    def _etf_monthly(data, config):
        etf = data.get("etf") or {}
        if etf.get("source_failed"):
            return build_section("ETF Monthly Performance", "Données ETF indisponibles ce mois.")
        rows = []
        for sym, t in etf.get("tickers", {}).items():
            if t.get("price") is not None:
                rows.append([sym, format_currency(t["price"]),
                             format_pct(t.get("pct_change", 0.0))])
        table = _table(["Ticker", "Prix", "Variation"], rows)
        prompt = f"Commentaire ETF mensuel ~150 mots. Tickers: {list(etf.get('tickers', {}).keys())}."
        body = synthesize_section(prompt, config=config, system=MONTHLY_SYSTEM_PROMPT)
        return build_section("ETF Monthly Performance", f"{table}\n\n{body}" if table else body)

    def _crypto_monthly(data, config):
        crypto = data.get("crypto") or {}
        if crypto.get("source_failed"):
            return build_section("Crypto Monthly", "Données crypto indisponibles ce mois.")
        rows = []
        for cid, c in crypto.get("coins", {}).items():
            if c.get("price") is not None:
                rows.append([c.get("symbol", cid), format_currency(c["price"]),
                             format_pct(c.get("pct_change_24h", 0.0))])
        table = _table(["Coin", "Prix", "24h"], rows)
        fg = crypto.get("fear_greed", {}) or {}
        prompt = (f"Commentaire crypto mensuel ~150 mots. Fear & Greed: "
                  f"{fg.get('label', 'n/a')} ({fg.get('value', 'n/a')}).")
        body = synthesize_section(prompt, config=config, system=MONTHLY_SYSTEM_PROMPT)
        return build_section("Crypto Monthly", f"{table}\n\n{body}" if table else body)

    def _pea_monthly(data, config):
        pea = data.get("pea") or {}
        if pea.get("source_failed"):
            return build_section("PEA Monthly", "Données PEA indisponibles ce mois.")
        rows = []
        for sym, p in pea.get("prices", {}).items():
            if p.get("price") is not None:
                rows.append([sym, format_currency(p["price"], symbol="€"),
                             format_pct(p.get("pct_change", 0.0))])
        table = _table(["Ticker", "Prix", "Variation"], rows)
        alert = ""
        if pea.get("eligibility_changed"):
            alert = "ALERTE — changement d'éligibilité PEA détecté ce mois.\n\n"
        prompt = "Commentaire PEA France mensuel ~150 mots."
        body = synthesize_section(prompt, config=config, system=MONTHLY_SYSTEM_PROMPT)
        out = f"{alert}{table}\n\n{body}" if table else f"{alert}{body}"
        return build_section("PEA Monthly", out)

    def _news_themes(data, config):
        news = data.get("news") or {}
        if news.get("source_failed"):
            return build_section("News & Themes", "Headlines indisponibles ce mois.")
        headlines = news.get("headlines", [])[:15]
        if not headlines:
            return build_section("News & Themes", "Aucun titre récent disponible.")
        bullets = "\n".join(f"- **{h.get('source', '?')}** — {h.get('title', 'titre')}" for h in headlines)
        prompt = f"Identifie les 3 thèmes financiers dominants du mois (~150 mots) à partir de ces titres :\n{bullets}"
        body = synthesize_section(prompt, config=config, system=MONTHLY_SYSTEM_PROMPT)
        return build_section("News & Themes", f"{bullets}\n\n{body}")

    def _forward_look(data, config):
        prompt = (
            "Rédige 'Forward Look' (~250 mots) — perspectives pour le mois à venir. "
            "Mentionne 3-4 points d'attention pour un investisseur particulier français."
        )
        body = synthesize_section(prompt, config=config, system=MONTHLY_SYSTEM_PROMPT, max_tokens=600)
        return build_section("Forward Look", body)

    def build_monthly_report(data: dict, config: Config) -> str:
        """Construit le Monthly Close (~2000 mots, 7 sections, ≥ 2 tableaux). Ne lève jamais."""
        try:
            sections = [
                _month_in_review(data, config),
                _macro_backdrop(data, config),
                _etf_monthly(data, config),
                _crypto_monthly(data, config),
                _pea_monthly(data, config),
                _news_themes(data, config),
                _forward_look(data, config),
            ]
            return "\n".join(sections)
        except Exception as e:
            logger.error("build_monthly_report failed: %s", e)
            return "\n".join(
                build_section(t, "[Section indisponible.]")
                for t in ("Month in Review", "Macro Backdrop", "ETF Monthly Performance",
                          "Crypto Monthly", "PEA Monthly", "News & Themes", "Forward Look")
            )
    ```

    **Étape B — Implémenter `reporters/dispatch.py`** :

    ```python
    """
    reporters/dispatch.py — Sélection de rapports selon la date du jour (REPT-04).

    Logique calendaire :
      - dernier jour du mois ET dimanche  → Monthly Close + Weekly Wrap (DEUX documents)
      - dernier jour du mois (semaine)    → Monthly Close
      - dimanche (sauf dernier du mois)   → Weekly Wrap
      - jours en semaine                  → Daily Report
    """
    from __future__ import annotations
    import calendar
    import datetime as _dt
    from typing import Dict
    from config import Config
    from reporters.daily import build_daily_report
    from reporters.weekly import build_weekly_report
    from reporters.monthly import build_monthly_report
    from logging_setup import get_logger

    logger = get_logger(__name__)


    def is_last_day_of_month(today: _dt.date) -> bool:
        """True si today est le dernier jour calendaire de son mois."""
        last = calendar.monthrange(today.year, today.month)[1]
        return today.day == last


    def is_sunday(today: _dt.date) -> bool:
        """True si today est un dimanche (weekday() == 6)."""
        return today.weekday() == 6


    def _safe_build(builder, name: str, data: dict, config: Config) -> str:
        """Wrapper qui ne lève jamais — encapsule l'échec d'un builder."""
        try:
            return builder(data, config)
        except Exception as e:
            logger.error("%s build failed: %s", name, e)
            return f"[{name} indisponible — erreur lors de la construction du rapport.]"


    def select_reports(today: _dt.date, data: dict, config: Config) -> Dict[str, str]:
        """
        Retourne un dict des rapports à émettre pour la date `today`.

        Args:
            today: date du jour (date.today() en production)
            data: dict produit par collect_all()
            config: Config

        Returns:
            dict avec une ou plusieurs clés parmi {"daily", "weekly", "monthly"}.
            REPT-04 : si today est le dernier jour du mois ET un dimanche, le dict
            contient à la fois "monthly" et "weekly".
        """
        last = is_last_day_of_month(today)
        sunday = is_sunday(today)
        result: Dict[str, str] = {}

        if last and sunday:
            # REPT-04 — les DEUX rapports sont produits dans le même run
            result["monthly"] = _safe_build(build_monthly_report, "Monthly Close", data, config)
            result["weekly"] = _safe_build(build_weekly_report, "Weekly Wrap", data, config)
        elif last:
            result["monthly"] = _safe_build(build_monthly_report, "Monthly Close", data, config)
        elif sunday:
            result["weekly"] = _safe_build(build_weekly_report, "Weekly Wrap", data, config)
        else:
            result["daily"] = _safe_build(build_daily_report, "Daily Report", data, config)

        return result
    ```
