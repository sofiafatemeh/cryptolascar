    ```python
    def collect_all(config: "Config") -> dict:
        """Orchestrate all 5 collectors. Catches per-collector exceptions.
        Returns combined data dict and metadata for downstream report builders.
        """
        results: dict = {}
        sources_ok: list[str] = []
        sources_failed: list[str] = []

        collectors_map = {
            "etf": collect_etf,
            "crypto": collect_crypto,
            "pea": collect_pea,
            "macro": collect_macro,
            "news": collect_news,
        }

        for name, collector_fn in collectors_map.items():
            try:
                data = collector_fn(config)
                results[name] = data
                if data.get("partial"):
                    sources_ok.append(f"{name}(partial)")
                else:
                    sources_ok.append(name)
                logger.info("Collector '%s' completed — partial=%s", name, data.get("partial", False))
            except Exception as e:
                logger.error("Collector '%s' failed with unhandled exception: %s", name, e)
                results[name] = {"error": str(e), "source_failed": True}
                sources_failed.append(name)

        results["_meta"] = {
            "sources_ok": sources_ok,
            "sources_failed": sources_failed,
            "collected_at": __import__("datetime").datetime.now(__import__("datetime").timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ"),
        }
        return results
    ```
