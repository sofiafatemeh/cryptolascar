    ```python
    from __future__ import annotations
    import json
    import time
    from datetime import datetime, timedelta, timezone
    from typing import Any

    import httpx

    from config import Config
    from db.cache import get_connection
    from logging_setup import get_logger

    logger = get_logger(__name__)

    FRED_SERIES = ["DGS10", "DGS2", "CPIAUCSL", "M2SL"]
    CACHE_SOURCE = "fred"
    CACHE_TTL_HOURS = 24
    FRED_SLEEP_SECONDS = 1.0
    FRED_BASE_URL = "https://api.stlouisfed.org/fred/series/observations"


    def _utcnow_iso() -> str:
        return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")

    def _expires_iso(hours: int) -> str:
        return (datetime.now(timezone.utc) + timedelta(hours=hours)).strftime("%Y-%m-%dT%H:%M:%SZ")

    def _get_cached(conn, source: str, symbol: str) -> dict | None:
        now = _utcnow_iso()
        row = conn.execute(
            "SELECT data_json FROM market_cache WHERE source=? AND symbol=? AND expires_at > ?",
            (source, symbol, now)
        ).fetchone()
        return json.loads(row["data_json"]) if row else None

    def _upsert_cache(conn, source: str, symbol: str, data: dict) -> None:
        conn.execute(
            "INSERT OR REPLACE INTO market_cache (source, symbol, data_json, fetched_at, expires_at) VALUES (?,?,?,?,?)",
            (source, symbol, json.dumps(data), _utcnow_iso(), _expires_iso(CACHE_TTL_HOURS))
        )
        conn.commit()

    def _fetch_fred_series(series_id: str, api_key: str) -> dict:
        """Fetch most recent observation for a FRED series. Raises on failure."""
        params = {
            "series_id": series_id,
            "api_key": api_key,
            "file_type": "json",
            "sort_order": "desc",
            "limit": 2,
        }
        resp = httpx.get(FRED_BASE_URL, params=params, timeout=15)
        observations = resp.json().get("observations", [])
        # Find most recent non-missing value ("." means data not yet available)
        for obs in observations:
            if obs["value"] != ".":
                return {"value": float(obs["value"]), "date": obs["date"], "series_id": series_id}
        raise ValueError(f"No valid observation found for {series_id}")

    def collect_macro(config: Config) -> dict:
        """Collect macro indicators from FRED API: DGS10, DGS2, CPIAUCSL, M2SL.
        Returns dict with 'series', 'fred_failed', 'partial', 'source_used'.
        Never raises — on failure returns partial dict with error flags.
        """
        if not config.fred_api_key:
            logger.warning("FRED_API_KEY not configured — skipping macro collection")
            return {"series": {}, "fred_failed": True, "partial": True, "source_used": CACHE_SOURCE}

        conn = get_connection(config.db_path)
        series_data: dict[str, Any] = {}
        fred_failed = False
        partial = False
        first_call = True

        for series_id in FRED_SERIES:
            cached = _get_cached(conn, CACHE_SOURCE, series_id)
            if cached:
                logger.debug("Cache hit for FRED series %s", series_id)
                series_data[series_id] = cached
                continue
            try:
                if not first_call:
                    time.sleep(FRED_SLEEP_SECONDS)
                first_call = False
                data = _fetch_fred_series(series_id, config.fred_api_key)
                time.sleep(FRED_SLEEP_SECONDS)
                _upsert_cache(conn, CACHE_SOURCE, series_id, data)
                logger.info("Fetched FRED %s: value=%.4f (date=%s)", series_id, data["value"], data["date"])
                series_data[series_id] = data
            except Exception as e:
                logger.error("FRED fetch failed for %s: %s", series_id, e)
                series_data[series_id] = {"value": None, "date": None, "series_id": series_id, "error": str(e)}
                partial = True

        conn.close()
        return {
            "series": series_data,
            "fred_failed": fred_failed,
            "partial": partial,
            "source_used": CACHE_SOURCE,
        }
    ```

    Note on sleep logic: Call time.sleep(FRED_SLEEP_SECONDS) after each successful FRED HTTP call to maintain 1s between requests. This ensures 4 series = 4 sleeps maximum.
